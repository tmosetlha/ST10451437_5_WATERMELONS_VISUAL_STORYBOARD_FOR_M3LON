// SCREEN 10: PROFILE + GAMIFICATION
M3LON's KEY DIFFERENTIATOR — none of the 3 researched apps (Spendee, Wallet, YNAB) have this!

▸ XP awarded via GamificationManager.kt on each expense save
▸ Badges stored in Room Badge table, unlocked by trigger conditions
▸ Streak counter = consecutive days with ≥1 expense logged
▸ CircularProgressBar custom view for XP bar
▸ GridLayout for badge display
▸ Settings menu: Light/Dark/System theme via AppCompatDelegate
GamificationManager.kt
object GamificationManager {
  fun onExpenseLogged(db: AppDatabase) {
    db.userDao().addXP(10) // +10 XP per entry
    db.userDao().incrementStreak()
    checkBadges(db)
  }

  private fun checkBadges(db: AppDatabase) {
    val user = db.userDao().getUser()
    if (user.streak >= 7)
      db.badgeDao().unlock("WEEK_WARRIOR")
    if (user.xp >= 1000)
      db.badgeDao().unlock("BUDGET_BOSS")
  }
}
